<script>

    // 价格计算逻辑

    function u(){

        let h = Math.max(1, parseInt(document.getElementById('h').value) || 1),

            db = document.getElementById('db').checked ? 10 : 0,

            e = document.getElementById('ht').checked ? 10 : (document.getElementById('o').checked ? 5 : 0),

            t = parseInt(document.getElementById('t').value);

        let total = (30 + db + e) * h + t;

        document.getElementById('tc').innerText = '$' + total;

        document.getElementById('fc').innerText = '$' + (total - 25);

    }

    u(); // 初始化计算


    // 一键截图逻辑

    function takeScreenshot() {

        const card = document.getElementById('captureCard');

        const overlay = document.getElementById('screenshotOverlay');

        const imgElement = document.getElementById('screenshotImage');


        // 使用 html2canvas 渲染图片

        html2canvas(card, {

            backgroundColor: "#fff5f7", // 保持背景颜色

            scale: 2, // 提高图片清晰度

            useCORS: true, // 允许跨域

            logging: false

        }).then(canvas => {

            // 将 canvas 转为 base64 图片地址

            imgElement.src = canvas.toDataURL("image/png");

            // 显示遮罩层

            overlay.style.display = 'flex';

        });

    }


    // 关闭遮罩层

    function closeOverlay() {

        document.getElementById('screenshotOverlay').style.display = 'none';

    }

</script>
